//
// Define the various device type values.  Note that values used by Microsoft
// Corporation are in the range 0-32767, and 32768-65535 are reserved for use
// by customers.
//
//
// Device code Mio GPS specific device driver
//
#define FILE_DEVICE_MIO_SYSTEM            0x00008000
#define FILE_DEVICE_MIO_BATTERY           0x00008001
#define FILE_DEVICE_MIO_BACKLIGHT         0x00008002
#define FILE_DEVICE_MIO_AUDIO             0x00008003
#define FILE_DEVICE_MIO_SDMMC             0x00008004
#define FILE_DEVICE_MIO_DISPLAY           0x00008005
#define FILE_DEVICE_MIO_POWER		  0x00008006
#define FILE_DEVICE_MIO_BLUETOOTH         0x00008007
#define FILE_DEVICE_MIO_TV                0x00008008
#define FILE_DEVICE_MIO_COMPASS           0x00008009
#define FILE_DEVICE_MIO_WALKITALKIE       0x0000800A


//
// Macro definition for defining IOCTL and FSCTL function control codes.  Note
// that function codes 0-2047 are reserved for Microsoft Corporation, and
// 2048-4095 are reserved for customers
//
//
// Function control code for Mio GPS specific functions
//
#define MIO_FCC_BASE			2048
#define MIO_FCC_BATTERY			MIO_FCC_BASE
#define MIO_FCC_BACKLIGHT 		MIO_FCC_BASE+10  
#define MIO_FCC_AUDIO			MIO_FCC_BASE+20
#define MIO_FCC_SDMMC			MIO_FCC_BASE+30
#define MIO_FCC_NAND			MIO_FCC_BASE+40
#define MIO_FCC_SERIAL			MIO_FCC_BASE+50
#define MIO_FCC_USB			MIO_FCC_BASE+60
#define MIO_FCC_TOUCHSCREEN		MIO_FCC_BASE+70
#define MIO_FCC_DISPLAY			MIO_FCC_BASE+80   
#define MIO_FCC_BUTTON			MIO_FCC_BASE+90
#define MIO_FCC_POWER			MIO_FCC_BASE+100
#define MIO_FCC_USB			MIO_FCC_BASE+110
#define MIO_FCC_BLUETOOTH		MIO_FCC_BASE+120  
#define MIO_FCC_TV 		        MIO_FCC_BASE+130  
#define MIO_FCC_SYS			MIO_FCC_BASE+900  // Mio specific system function 

//
// IO control definition for Mio specific functions
//  
// Pattern for IOCTL defininition:  IOCTL_MIO_DEVICETYPE_FUNCTION(V+N)
//
// -------------- for system function -------------------------------------------------------
#define IOCTL_MIO_SYS_SWITCHOFF_DEVICE	 CT_CODE(FILE_DEVICE_MIO_SYSTEM,   MIO_FCC_SYS+1,         METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_SYS_SET_DVS_CODE	 CT_CODE(FILE_DEVICE_MIO_SYSTEM,   MIO_FCC_SYS+2,         METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_SYS_CLEAR_REGISTRY     CT_CODE(FILE_DEVICE_MIO_SYSTEM,   MIO_FCC_SYS+3,         METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_SYS_GET_CEIMG_VERSION	 CT_CODE(FILE_DEVICE_MIO_SYSTEM,   MIO_FCC_SYS+4,         METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_SYS_GET_CEIMG_CHECKSUM CT_CODE(FILE_DEVICE_MIO_SYSTEM,   MIO_FCC_SYS+5,         METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_SYS_GET_UBOOT_CHECKSUM CT_CODE(FILE_DEVICE_MIO_SYSTEM,   MIO_FCC_SYS+6,         METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_SYS_GET_BTMAC_ADDR	 CT_CODE(FILE_DEVICE_MIO_SYSTEM,   MIO_FCC_SYS+7,         METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_SYS_IBOOT   	         CT_CODE(FILE_DEVICE_MIO_SYSTEM,   MIO_FCC_SYS+8,         METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_SYS_IBOOT_SD           CT_CODE(FILE_DEVICE_MIO_SYSTEM,   MIO_FCC_SYS+9,         METHOD_BUFFERED, FILE_ANY_ACCESS)
// -------------- for backlight control function --------------------------------------------
#define IOCTL_MIO_BKL_GET_BRIGHTNESS	 CT_CODE(FILE_DEVICE_MIO_BACKLIGHT,MIO_FCC_BACKLIGHT+1,   METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_BKL_SET_BRIGHTNESS	 CT_CODE(FILE_DEVICE_MIO_BACKLIGHT,MIO_FCC_BACKLIGHT+2,   METHOD_BUFFERED, FILE_ANY_ACCESS)
// -------------- for screen control function -----------------------------------------------
#define IOCTL_MIO_DISP_OFF_SCREEN	 CT_CODE(FILE_DEVICE_MIO_DISPLAY,  MIO_FCC_DISPLAY+1,     METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_DISP_ON_SCREEN	 CT_CODE(FILE_DEVICE_MIO_DISPLAY,  MIO_FCC_DISPLAY+2,     METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_DISP_GET_SCREEN_STATUS CT_CODE(FILE_DEVICE_MIO_DISPLAY,  MIO_FCC_DISPLAY+3,     METHOD_BUFFERED, FILE_ANY_ACCESS)
// -------------- for timer control function -------------------------------------------------
#define IOCTL_MIO_POWER_SET_SUSPEND_TIMER     CT_CODE(FILE_DEVICE_MIO_POWER,   MIO_FCC_POWER+1,    METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_POWER_RESET_SUSPEND_TIMER   CT_CODE(FILE_DEVICE_MIO_POWER,   MIO_FCC_POWER+2,    METHOD_BUFFERED, FILE_ANY_ACCESS)
#define IOCTL_MIO_POWER_ENABLE_SUSPEND_TIMER  CT_CODE(FILE_DEVICE_MIO_POWER,   MIO_FCC_POWER+3,    METHOD_BUFFERED, FILE_ANY_ACCESS)
